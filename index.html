Developer API

Integrate voice generation in minutes

Build powerful voice experiences using our simple and scalable API. From real-time speech to voice cloning — everything is just a few lines of code away.

# Generate speech with CodeVoice API
curl -X POST https://api.codevoice.io/v1/tts \
  -H "Authorization: Bearer <your-api-key>" \
  -H "Content-Type: application/json" \
  -d '{
    "text": "Welcome to CodeVoice",
    "voice": "amy",
    "speed": 1.0,
    "format": "wav"
  }'
const response = await fetch("https://api.codevoice.io/v1/tts", {
  method: "POST",
  headers: {
    "Authorization": "Bearer <your-api-key>",
    "Content-Type": "application/json"
  },
  body: JSON.stringify({
    text: "Welcome to CodeVoice",
    voice: "amy",
    speed: 1.0,
    format: "wav"
  })
});

const audio = await response.blob();
import requests

response = requests.post(
    "https://api.codevoice.io/v1/tts",
    headers={
        "Authorization": "Bearer <your-api-key>",
        "Content-Type": "application/json"
    },
    json={
        "text": "Welcome to CodeVoice",
        "voice": "amy",
        "speed": 1.0,
        "format": "wav"
    }
)

audio = response.content
